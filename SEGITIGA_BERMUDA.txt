#include<stdio.h>
#include<math.h>
int main()
{
    int n = 5;
    for(int i=1;i<=n;i++){
        for(int j=(2*n);j>=0;j--){
            printf(" * ");
        }
        printf("\n");
    }
    return 0;
}
